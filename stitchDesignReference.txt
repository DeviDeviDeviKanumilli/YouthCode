<script>
    const sheet = document.getElementById('bottom-sheet');
    const dragHandle = document.getElementById('drag-handle');
    const headerArea = document.getElementById('header-area');
    const sheetContent = document.getElementById('sheet-content');

    let isDragging = false;
    let startY = 0;
    let initialTranslateY = 0;
    let currentTranslateY = 0;
    
    // Configurable thresholds (as % of screen height)
    // 0 = top of screen, 100 = bottom
    // Compact: shows ~35-40% of sheet height (leaving 60-65% hidden)
    // Expanded: shows ~85% of sheet height (leaving 0-5% hidden)
    const compactPos = 55; // % hidden
    const expandedPos = 5;  // % hidden
    const screenHeight = window.innerHeight;

    // Initialize position
    function initSheet() {
        currentTranslateY = (compactPos / 100) * (screenHeight * 0.85); 
        // Note: sheet height is 85vh, so we move it relative to that
        setSheetTransform(compactPos);
    }

    function setSheetTransform(percentHidden) {
        // We translate the sheet based on its height (85vh)
        // 0% hidden means Y = 0
        // 100% hidden means Y = 100% of sheet height
        sheet.style.transform = `translateY(${percentHidden}%)`;
    }

    function onDragStart(e) {
        isDragging = true;
        startY = e.type === 'touchstart' ? e.touches[0].clientY : e.clientY;
        
        // Disable transitions during manual drag
        sheet.classList.remove('sheet-transition');
        
        // Get current transform value
        const style = window.getComputedStyle(sheet);
        const matrix = new WebKitCSSMatrix(style.transform);
        initialTranslateY = matrix.m42;
    }

    function onDragMove(e) {
        if (!isDragging) return;
        
        const currentY = e.type === 'touchmove' ? e.touches[0].clientY : e.clientY;
        const deltaY = currentY - startY;
        let newTranslateY = initialTranslateY + deltaY;

        // Constraints: Don't drag above the top limit (expandedPos) 
        // or too far below compact
        const maxTranslate = (screenHeight * 0.85); 
        const minTranslate = (expandedPos / 100) * (screenHeight * 0.85);

        if (newTranslateY < minTranslate) {
            newTranslateY = minTranslate - Math.pow(minTranslate - newTranslateY, 0.7); // Rubber banding
        }

        sheet.style.transform = `translateY(${newTranslateY}px)`;
        currentTranslateY = newTranslateY;
    }

    function onDragEnd() {
        if (!isDragging) return;
        isDragging = false;
        
        sheet.classList.add('sheet-transition');
        
        const sheetHeight = screenHeight * 0.85;
        const compactPx = (compactPos / 100) * sheetHeight;
        const expandedPx = (expandedPos / 100) * sheetHeight;
        
        // Logic to snap to nearest state
        const midPoint = (compactPx + expandedPx) / 2;
        
        if (currentTranslateY < midPoint) {
            setSheetTransform(expandedPos);
            currentTranslateY = expandedPx;
            sheetContent.style.overflowY = "auto";
        } else {
            setSheetTransform(compactPos);
            currentTranslateY = compactPx;
            sheetContent.style.overflowY = "hidden";
        }
    }

    // Event Listeners
    [dragHandle, headerArea].forEach(el => {
        el.addEventListener('mousedown', onDragStart);
        el.addEventListener('touchstart', onDragStart, { passive: true });
    });

    window.addEventListener('mousemove', onDragMove);
    window.addEventListener('touchmove', onDragMove, { passive: false });
    window.addEventListener('mouseup', onDragEnd);
    window.addEventListener('touchend', onDragEnd);

    // Click on header to toggle
    headerArea.addEventListener('click', () => {
        sheet.classList.add('sheet-transition');
        const sheetHeight = screenHeight * 0.85;
        const compactPx = (compactPos / 100) * sheetHeight;
        
        if (Math.abs(currentTranslateY - compactPx) < 10) {
            setSheetTransform(expandedPos);
            currentTranslateY = (expandedPos / 100) * sheetHeight;
            sheetContent.style.overflowY = "auto";
        } else {
            setSheetTransform(compactPos);
            currentTranslateY = compactPx;
            sheetContent.style.overflowY = "hidden";
        }
    });

    // Handle map clicks to collapse
    document.querySelector('main').addEventListener('click', () => {
        sheet.classList.add('sheet-transition');
        setSheetTransform(compactPos);
        currentTranslateY = (compactPos / 100) * (screenHeight * 0.85);
        sheetContent.style.overflowY = "hidden";
    });

    // Initialize
    initSheet();
</script>
